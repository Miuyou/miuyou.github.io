	dp[0][0]=1;
	for(int i=1;i<=n;++i){
		if(a[i]>=b[i])
		for(int j=0;j<=min(n,k);++j){
			SpringWait x;
			x=(c[i]+j+1);
			dp[i][j]+=x*dp[i-1][j];
			if(!j)continue;
			x=(i-c[i]-j);
			if(i-c[i]-j)
			dp[i][j]+=x*dp[i-1][j-1];
		}else{
			for(int j=1;j<=min(n,k);++j){
				SpringWait x;
				x=(d[i]+i-j+1);
				dp[i][j]+=x*dp[i-1][j-1];
				x=(j-d[i]);
				if(j-d[i]>0)
				dp[i][j]+=x*dp[i-1][j];
			}
		}
	}
