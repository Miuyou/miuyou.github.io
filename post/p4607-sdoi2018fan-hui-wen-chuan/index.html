inline bool miller(ll x){
    if(x<2)return 0;
    if(x==2||x==3)return 1;
    ll y=x-1,r=0;
    while(!(y&1))++r,y>>=1;
    for(int i=0;i<=6;++i){
        ll a=prim[i]%(x-2)+2;
        ll z=ksm(a,y,x);
        if(z==1||z==x-1)continue;
        for(int j=0;j<r-1;++j){
            z=mul(z,z,x);
            if(z==x-1)break;
        }
        if(z!=x-1)return 0;
    }return 1;
}
inline ll rho(ll x){
    ll s=0,t=0,z=1ll*rand()%(x-1)+1,val=1;
    int num=0,goal=1;
    for(;;goal<<=1,s=t,val=1){
        for(num=1;num<=goal;++num){
            t=F(t,z,x);
            val=mul(val,abs(t-s),x);
            if(num%127==0){
                ll k=gcd(val,x);
                if(k>1)return k;
            }
        }
        ll k=gcd(val,x);
        if(k>1)return k;
    }
}

inline void dp(ll x,ll f, ll d){

        if(i==num[x])dp(x+1,f,d*h);

        h=h*val[x];
    }
}
